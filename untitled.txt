<body>

<div class="container">
    <h2>စာရင်းမှတ်စနစ် PRO</h2>
    <input type="text" id="userInput" placeholder="ဥပမာ - 00 1000" onkeypress="handleEnter(event)">
    <button onclick="addData()">ပေါင်းထည့်မည်</button>
    <button class="reset-btn" onclick="clearData()">စာရင်းအားလုံးဖျက်မည်</button>

    <div id="list" style="margin-top: 20px; max-height: 300px; overflow-y: auto;"></div>
    <div class="total" id="totalDisplay">စုစုပေါင်း - 0 ကျပ်</div>
</div>

<script>
    let dataStorage = JSON.parse(localStorage.getItem('myLists')) || [];

    // စဖွင့်ချင်းရှိတာတွေ ပြပေးရန်
    renderList();

    function handleEnter(event) {
        if (event.key === "Enter") addData();
    }

    function addData() {
        const input = document.getElementById('userInput');
        const value = input.value.trim();
        const parts = value.split(" ");

        if (parts.length === 2) {
            const item = { number: parts[0], amount: parseInt(parts[1]) };
            if (!isNaN(item.amount)) {
                dataStorage.push(item);
                localStorage.setItem('myLists', JSON.stringify(dataStorage));
                input.value = "";
                renderList();
            }
        }
    }

    function renderList() {
        const list = document.getElementById('list');
        const totalDisplay = document.getElementById('totalDisplay');
        list.innerHTML = "";
        let total = 0;

        dataStorage.slice().reverse().forEach(item => {
            const div = document.createElement('div');
            div.className = 'item';
            div.innerHTML = `<span>နံပါတ်: <b>${item.number}</b></span> <span>${item.amount.toLocaleString()} ကျပ်</span>`;
            list.appendChild(div);
            total += item.amount;
        });

        totalDisplay.innerText = `စုစုပေါင်း - ${total.toLocaleString()} ကျပ်`;
    }

    function clearData() {
        if(confirm("စာရင်းအားလုံးကို ဖျက်မှာ သေချာပါသလား?")) {
            dataStorage = [];
            localStorage.removeItem('myLists');
            renderList();
        }
    }
</script>
